class Solution{
    
  public int mySqrt(int x){

    if (x == 0)  
   {
     return 0;
   }   
 
   long l = 1;
   long h = x, ans = 1;

while(1 <= h)
 {
	long mid = (1 + h) / 2;

  if(mid * mid == x){
  ans = mid;
  break;
   }
  else if(mid * mid > x)  {
      h = mid - 1;
   }

  else{
     ans = mid;
     l = mid + 1;
   }

 }

  return (int) ans;
     }
}
